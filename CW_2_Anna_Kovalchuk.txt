#include <stdio.h>

int main() {
  
    double x, y;
    double result;

    double test_values[3][2] = {{1, 1}, {2, 2}, {0, 0}};

    printf("Rosenbrock function results:\n");
    for (int i = 0; i < 3; ++i) {
        x = test_values[i][0];
        y = test_values[i][1];
        result = 100 * (x * x - y) * (x * x - y) + (x - 1) * (x - 1);
        printf("Rosenbrock2d(%.2f, %.2f) = %.2f\n", x, y, result);
    }

    return 0;
}

#include <stdio.h>
#include <math.h>

int main() {
  
    double Ax, Ay, Bx, By, Cx, Cy;

    
    printf("Enter the coordinates of point A (x y): ");
    scanf("%lf %lf", &Ax, &Ay);

    printf("Enter the coordinates of point B (x y): ");
    scanf("%lf %lf", &Bx, &By);

    printf("Enter the coordinates of point C (x y): ");
    scanf("%lf %lf", &Cx, &Cy);

 
    double a = sqrt((Bx - Cx) * (Bx - Cx) + (By - Cy) * (By - Cy));
    double b = sqrt((Ax - Cx) * (Ax - Cx) + (Ay - Cy) * (Ay - Cy));
    double c = sqrt((Ax - Bx) * (Ax - Bx) + (Ay - By) * (Ay - By));

        double s = (a + b + c) / 2;

    double area = sqrt(s * (s - a) * (s - b) * (s - c));

       printf("The area of the triangle with vertices A(%.2f, %.2f), B(%.2f, %.2f), and C(%.2f, %.2f) is %.2f\n",
           Ax, Ay, Bx, By, Cx, Cy, area);

    return 0;
}
