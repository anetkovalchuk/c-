#include <stdio.h>

int main() {

    int result1 = 2 + 31;
    int result2 = 45 * 54 - 11;
    int result3 = 15 / 4;
    double result4 = 15.0 / 4;
    int result5 = 67 % 5;
    double result6 = (2 * 45.1 + 3.2) / 2;

    
    printf("2 + 31 = %d\n", result1);
    printf("45 * 54 - 11 = %d\n", result2);
    printf("15 / 4 = %d\n", result3);
    printf("15.0 / 4 = %.2f\n", result4);
    printf("67 %% 5 = %d\n", result5);
    printf("(2 * 45.1 + 3.2) / 2 = %.2f\n", result6);

    return 0;
}

// 1.7
#include <stdio.h>
#include <math.h>

int main() {
    double x;

    printf("Enter the number: ");
    scanf("%lf", &x);

    // modf- озділяє число на цілу та дробову частини
    // int_part - ціла частина, frac_part - дробова
    double int_part;
    double frac_part = modf(x, &int_part);

    // ceil - повертає найменше ціле число
    double next_up = ceil(x);

    // floor найбільше ціле число
    double prev_down = floor(x);


    double rounded = round(x);


    printf("Entered number : %.2lf\n", x);
    printf("Whole part: %.0lf\n", int_part);
    printf("fractoinal part: %.2lf\n", frac_part);
    printf("The smallest integer: %.0lf\n", next_up);
    printf("The largest integer: %.0lf\n", prev_down);
    printf("Rounded value: %.0lf\n", rounded);

    return 0;
}
