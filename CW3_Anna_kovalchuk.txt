void task1() {
    int a, b, c, sum, reverse;
    unsigned number;
    printf("Enter a number: ");
    scanf("%u", &number);

    if (number >= 100 && number <= 999) {
        a = number % 10;
        b = (number / 10) % 10;
        c = (number / 100);

        printf(" a = %d\n", a);
        printf(" b = %d\n", b);
        printf(" c = %d\n", c);

        sum = a + b + c;
        reverse = a * 100 + 10 * b + c;

        printf(" sum = %d\n", sum);
        printf(" reverse = %d\n", reverse);
    }
}
